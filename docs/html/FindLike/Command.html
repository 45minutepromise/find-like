def initialize(argv)
  @p = argv && argv.flag?('P') | false
  @l = argv && argv.flag?('L') | false
  @exclude_dir = argv.option('exclude-dir')
  @path = argv.option('path')
  @type = argv.option('type')
  @name = argv.option('name')
  @rname = argv.option('rname')
  # validate_options
  super
end
